1function x_hat_pos = state_estimation(wheel, camera, u)
 2% Camera defaults to 0
 3% when we get a new camera value, reset kalman filter and recalcultate
 4persistent x_hat_k1_k1;
 5persistent P_k1_k1;
 6
 7% global F_k; % A
 8% global B_k; % B
 9% global H_k; % C
10% global Q_k; % Covariance of process noise
11% global R_k; % Covariance of observation noise
12% global I;
13
14coder.extrinsic('evalin');
15F_k = zeros(15,15);
16B_k = zeros(15,15);
17H_k = zeros(7,15);
18Q_k = zeros(15,15);
19R_k = zeros(15,15);
20F_k = evalin('base', 'F_k');
21B_k = evalin('base', 'B_k');
22H_k = evalin('base', 'H_k');
23Q_k = evalin('base', 'Q_k');
24R_k = evalin('base', 'R_k');
25
26
27if isempty(x_hat_k1_k1)
28    x_hat_k1_k1 = zeros(15,1);
29end
30
31if isempty(P_k1_k1)
32    P_k1_k1 = zeros(15, 15);
33end
34
35% Output data from plant
36z_k = [camera; wheel];
37u = [0;0;0;u;0;0;0;0;0;0;0;0];
38% Predict where we should be this time step
39x_hat_k_k1 = F_k*x_hat_k1_k1 + B_k*u;
40P_k_k1 = F_k*P_k1_k1*F_k' + Q_k;
41
42% Get error between predicted and actual
43y_tilda_k = z_k - H_k*x_hat_k_k1;
44S_k = R_k + H_k*P_k_k1*H_k';
45
46% Get optimal state estimator gain
47K_k = P_k_k1*H_k'*pinv(S_k);
48
49% Update state
50x_hat_k_k = x_hat_k_k1 + K_k*y_tilda_k;
51
52% Update final covariance / better output
53P_k_k = (I - K_k*H_k)*P_k_k1*(I - K_k*H_k)' + K_k*R_k*K_k';
54y_tilda_k_k = z_k - H_k*x_hat_k_k;
55
56x_hat_pos = x_hat_k_k(1:3);
57
58% Setup P for the next iteration
59P_k1_k1 = P_k_k;
60x_hat_k1_k1 = x_hat_k_k;
61end
